from django.shortcuts import render from .models import TableModel def chart_view(request): # 获取URL参数 selected_deal = request.GET.get('deal', 'All') selected_lp = request.GET.get('lp', 'All') # 获取Deal和LP的所有唯一值用于下拉菜单 deals = TableModel.objects.values_list('Deal', flat=True).distinct() lps = TableModel.objects.values_list('LP', flat=True).distinct() # 初始化查询集 queryset = TableModel.objects.all() # 应用筛选条件，除非选择了“All” if selected_deal != 'All': queryset = queryset.filter(Deal=selected_deal) if selected_lp != 'All': queryset = queryset.filter(LP=selected_lp) # 准备数据 data = list(queryset.values('Date', 'Data1', 'Data2', 'Data3', 'Data4').order_by('Date')) context = { 'deals': deals, 'lps': lps, 'selected_deal': selected_deal, 'selected_lp': selected_lp, 'data': data, } return render(request, 'your_template.html', context) //Echarts example var rawData = {{ data|safe }}; // 将数据按Data1的值分组 var groupedData = {}; rawData.forEach(function(item) { if (!groupedData[item.Data1]) { groupedData[item.Data1] = []; } groupedData[item.Data1].push({date: item.Date, value: item.Data2}); }); // 提取并去重所有日期，确保已排序 var dates = rawData.map(function(item) { return item.Date; }); dates = [...new Set(dates)].sort(function(a, b) { return new Date(a) - new Date(b); }); // 补充缺失的日期数据为null Object.keys(groupedData).forEach(function(key) { var fullData = []; dates.forEach(function(date) { var foundItem = groupedData[key].find(item => item.date === date); if (foundItem) { fullData.push(foundItem.value); } else { // 对于缺失的日期插入null fullData.push(null); } }); // 更新groupedData为包含所有日期的完整数据 groupedData[key] = fullData; }); // 创建Echarts系列数组 var series = Object.keys(groupedData).map(function(key) { return { name: key, type: 'line', // 直接使用完整数据，已经包含null值 data: groupedData[key] }; }); // 使用处理好的dates和series配置Echarts var option = { tooltip: { trigger: 'axis' }, xAxis: { type: 'category', data: dates, }, yAxis: { type: 'value' }, series: series, };
